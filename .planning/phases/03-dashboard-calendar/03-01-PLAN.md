```typescript
import { headers } from "next/headers";
import { auth } from "@/lib/auth";
import { getDashboardData } from "@/lib/queries/dashboard";
import { nowColombia } from "@/lib/utils/dates";
import { SummaryCards } from "@/components/dashboard/summary-cards";
import { CalendarGrid } from "@/components/dashboard/calendar-grid";
import { EmptyState } from "@/components/dashboard/empty-state";

export const metadata = {
  title: "Panel de control | Calendario Tributario",
};

export default async function DashboardPage() {
  const session = await auth.api.getSession({ headers: await headers() });
  const { obligations } = await getDashboardData(session!.user.id);
  const todayIso = nowColombia().toISOString();

  if (obligations.length === 0) {
    return <EmptyState type="no-obligations" />;
  }

  return (
    <div className="space-y-6">
      <h1 className="text-2xl font-bold tracking-tight">Panel de control</h1>
      <SummaryCards obligations={obligations} />
      <CalendarGrid obligations={obligations} todayIso={todayIso} />
      {/* ObligationsTable will be added in Plan 03-02 */}
    </div>
  );
}
```

The page fetches all obligations via `getDashboardData`, computes Colombia "today", and passes serialized data to client components. The `todayIso` prop ensures calendar "today" is always Colombia time, not browser/server time.

CRITICAL: session! is safe because AppShell already gates unauthenticated access with redirect. The getDashboardData function handles missing empresa gracefully.
